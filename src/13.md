# Translate labels

Load the labels

```php
$this->pageRenderer->addInlineLanguageLabelFile('EXT:lit_demo/Resources/Private/Language/locallang.xlf');
```

JavaScript

```js
import {LitElement, html, nothing} from 'lit';
import {lll} from '@typo3/core/lit-helper.js';

export class TranslateExample extends LitElement {
    render() {
        return html`<h1>${lll('label-key')}</h1>`;
    }
}

customElements.define('translate-example', TranslateExample)
```
